```python
# src/dataforge/config/schema.py
from __future__ import annotations

from typing import Annotated, Literal

from pydantic import BaseModel, Field


class LLMConfig(BaseModel):
    provider: Literal["openai", "vllm", "anthropic"]
    model: str
    base_url: str | None = None
    api_key: str | None = None
    rpm_limit: int = 60
    tpm_limit: int = 100_000
    concurrency: int = 50
    generation_kwargs: dict = {}


class SourceConfig(BaseModel):
    type: Literal["jsonl"]
    path: str


class SinkConfig(BaseModel):
    path: str
    checkpoint_dir: str = "./.dataforge_runs"


class GenerateStepConfig(BaseModel):
    step: Literal["generate"]
    strategy: Literal["evol-instruct"]
    depth: int = 3
    mutation_types: list[str] | None = None
    require_reasoning: bool = False
    require_json: bool = False
    llm: LLMConfig


class EvaluateStepConfig(BaseModel):
    step: Literal["evaluate"]
    evaluator: Literal["llm-judge", "regex-filter"]
    llm: LLMConfig | None = None
    criteria: str = "helpfulness"
    threshold: float = 4.0
    blacklist_patterns: list[str] = []
    require_json: bool = False


StepConfig = Annotated[
    GenerateStepConfig | EvaluateStepConfig,
    Field(discriminator="step"),
]


class ForgeConfig(BaseModel):
    name: str
    source: SourceConfig
    pipeline: list[StepConfig]
    sink: SinkConfig
```

```python
# src/dataforge/config/loader.py
from __future__ import annotations

import os

import yaml

from dataforge.clients import OpenAIClient, vLLMClient
from dataforge.clients.base import BaseLLMClient
from dataforge.config.schema import (
    EvaluateStepConfig,
    ForgeConfig,
    GenerateStepConfig,
    LLMConfig,
)
from dataforge.evaluators import LLMJudge, RegexFilter
from dataforge.evaluators.base import BaseEvaluator
from dataforge.pipeline import Pipeline
from dataforge.strategies import EvolInstruct
from dataforge.strategies.base import BaseStrategy

_ENV_KEYS: dict[str, str | None] = {
    "openai": "OPENAI_API_KEY",
    "anthropic": "ANTHROPIC_API_KEY",
    "vllm": None,
}


def load_config(path: str) -> ForgeConfig:
    """Read a YAML file and return a validated ForgeConfig."""
    with open(path, "r", encoding="utf-8") as f:
        data = yaml.safe_load(f)
    return ForgeConfig.model_validate(data)


def build_client(cfg: LLMConfig) -> BaseLLMClient:
    """Instantiate the correct LLM client from a LLMConfig."""
    api_key = cfg.api_key
    if api_key is None:
        env_var = _ENV_KEYS.get(cfg.provider)
        if env_var:
            api_key = os.environ.get(env_var)

    if cfg.provider == "vllm":
        return vLLMClient(
            model=cfg.model,
            base_url=cfg.base_url or "http://localhost:8000/v1",
            rpm_limit=cfg.rpm_limit,
            tpm_limit=cfg.tpm_limit,
            generation_kwargs=cfg.generation_kwargs,
        )
    return OpenAIClient(
        model=cfg.model,
        api_key=api_key,
        base_url=cfg.base_url,
        rpm_limit=cfg.rpm_limit,
        tpm_limit=cfg.tpm_limit,
        generation_kwargs=cfg.generation_kwargs,
    )


def build_strategy(step: GenerateStepConfig) -> BaseStrategy:
    """Build an EvolInstruct strategy from a GenerateStepConfig."""
    llm = build_client(step.llm)
    return EvolInstruct(
        llm=llm,
        depth=step.depth,
        mutation_types=step.mutation_types,
        require_reasoning=step.require_reasoning,
        require_json=step.require_json,
    )


def build_evaluator(step: EvaluateStepConfig) -> BaseEvaluator:
    """Build a RegexFilter or LLMJudge evaluator from an EvaluateStepConfig."""
    if step.evaluator == "regex-filter":
        return RegexFilter(
            blacklist_patterns=step.blacklist_patterns,
            require_json=step.require_json,
        )
    if step.evaluator == "llm-judge":
        if step.llm is None:
            raise ValueError(
                "evaluator 'llm-judge' requires an 'llm' config block"
            )
        return LLMJudge(
            llm=build_client(step.llm),
            criteria=step.criteria,
            threshold=step.threshold,
        )
    raise ValueError(f"Unknown evaluator: {step.evaluator!r}")


def build_pipeline(config: ForgeConfig) -> tuple[Pipeline, str, str, int]:
    """Assemble a Pipeline from a ForgeConfig.

    Returns:
        (pipeline, input_path, output_path, concurrency)
    """
    strategy: BaseStrategy | None = None
    evaluators: list[BaseEvaluator] = []
    concurrency = 50

    for step in config.pipeline:
        if step.step == "generate":
            strategy = build_strategy(step)
            concurrency = step.llm.concurrency
        elif step.step == "evaluate":
            evaluators.append(build_evaluator(step))

    if strategy is None:
        raise ValueError(
            "Pipeline config must contain at least one 'generate' step"
        )

    pipeline = Pipeline(
        strategy=strategy,
        evaluators=evaluators,
        checkpoint_dir=config.sink.checkpoint_dir,
    )
    return pipeline, config.source.path, config.sink.path, concurrency
```

```python
# src/dataforge/cli.py
from __future__ import annotations

import argparse
import asyncio
import sys


def main() -> None:
    parser = argparse.ArgumentParser(
        prog="dataforge",
        description="DataForge — high-concurrency LLM data synthesis pipeline",
    )
    subparsers = parser.add_subparsers(dest="command", required=True)

    run_parser = subparsers.add_parser(
        "run", help="Run a synthesis pipeline from a YAML config file"
    )
    run_parser.add_argument(
        "config", metavar="CONFIG", help="Path to YAML config file"
    )
    run_parser.add_argument(
        "--dry-run",
        action="store_true",
        help="Validate config and exit without running the pipeline",
    )

    args = parser.parse_args()
    if args.command == "run":
        _run(args.config, dry_run=args.dry_run)


def _run(config_path: str, dry_run: bool = False) -> None:
    from dataforge.config import build_pipeline, load_config

    try:
        config = load_config(config_path)
    except Exception as e:
        print(
            f"Error: failed to parse config '{config_path}': {e}",
            file=sys.stderr,
        )
        sys.exit(1)

    if dry_run:
        print(f"Config '{config.name}' is valid.")
        return

    pipeline, input_path, output_path, concurrency = build_pipeline(config)
    asyncio.run(
        pipeline.run(
            input_path=input_path,
            output_path=output_path,
            concurrency=concurrency,
        )
    )
```
